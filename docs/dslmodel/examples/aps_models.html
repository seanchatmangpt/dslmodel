147def run_participants_concurrently():
148    """Create 5 participants concurrently using the run_dsls function."""
149    # Prepare tasks for creating 5 different participants concurrently
150    from dslmodel.template import render
151
152    tasks = [
153        (Participant, render("Generate participant {{fake_name()}}, {{ fake_job() }} for the board meeting. "
154                             "Start time {{ fake_time() }}")) for i in range(5)
155    ]
156
157    # Run tasks concurrently
158    results = run_dsls(tasks, max_workers=5)
159
160    # Output the generated participants
161    for i, result in enumerate(results):
162        print(f"Participant {i+1}: {result}")
