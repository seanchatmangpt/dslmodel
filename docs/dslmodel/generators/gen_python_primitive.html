117class GenPrimitiveModule:
118    """A generic module for generating and validating primitive types from prompts."""
119
120    def __init__(self, primitive_type, generate_sig: Type[Signature], error_sig: Type[Signature], verbose=False):
121        """
122        Initialize the module with the expected primitive type and the relevant DSPy signatures.
123
124        Args:
125        - primitive_type: The expected Python primitive type (e.g., int, float, str, etc.).
126        - generate_sig: The DSPy signature class for generating the primitive.
127        - error_sig: The DSPy signature class for handling errors during generation.
128        - verbose: Whether to enable verbose logging.
129        """
130        self.primitive_type = primitive_type
131        self.generate_sig = generate_sig
132        self.error_sig = error_sig
133        self.verbose = verbose
134        self.output_key = "output"
135
136        # Initialize DSPy ChainOfThought modules for generation and error correction
137        self.generate = ChainOfThought(generate_sig)
138        self.correct_generate = ChainOfThought(error_sig)
139        self.validation_error = None
140
141    def eval_output(self, output: str):
142        """
143        Attempt to evaluate the generated output string into the correct Python primitive type.
144        """
145        return eval_output_str(output)
146
147    def validate_output(self, output) -> bool:
148        """
149        Validates whether the output matches the expected primitive type.
150        """
151        is_valid = isinstance(output, self.primitive_type)
152        Assert(is_valid, f"Expected {self.primitive_type.__name__}, but got {type(output).__name__}.")
153        return is_valid
154
155    def forward(self, prompt: str):
156        """
157        Handles generation, validation, correction, and returns the final output.
158        """
159        # Generate initial output
160        generated_output = self.generate(prompt=prompt)
161        output = generated_output.get(self.output_key)
162
163        # Attempt to evaluate the output to the expected primitive type
164        evaluated_output = self.eval_output(output)
165
166        # Validate the output
167        try:
168            if self.validate_output(evaluated_output):
169                return evaluated_output
170        except AssertionError as e:
171            logger.warning(f"Validation failed: {e}. Trying correction...")
172
173        # Attempt to correct the output
174        corrected_output = self.correct_generate(prompt=prompt, error="Validation failed", generated_output=output)
175        corrected_evaluated_output = self.eval_output(corrected_output.get(self.output_key))
176
177        # Validate corrected output
178        Assert(self.validate_output(corrected_evaluated_output),
179               f"Correction failed to generate valid {self.primitive_type.__name__}.")
180        return corrected_evaluated_output
181
182    def __call__(self, prompt: str):
183        return self.forward(prompt)
